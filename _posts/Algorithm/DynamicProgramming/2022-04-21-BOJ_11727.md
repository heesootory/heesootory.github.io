> 타일링1 문제에서 한가지의 경우만 더 추가된 경우이고,
>  
> 점화식은 d[n] = d[n-1] +d[n-2] + d[n-2]


#### C++ 코드(bottom-top)
```cpp
#include <iostream>
using namespace std;

int d[1001] = { 0,1,3 };

int main() {
	int n; 
	cin >> n;

	for (int i = 3; i <= n; i++) {
		d[i] = (d[i - 1] + d[i - 2] * 2) % 10007;
	}

	cout << d[n];

}


```

using System.Text;
using System;
using static System.Console;

class Program
{
    static long[] arr = new long[1001];
    public static void Main(string[] args)
    {
        arr[0] = 0; arr[1] = 1; arr[2] = 3;

        int n = int.Parse(ReadLine());
        for( int i = 3; i<=n; i++)
        {
            arr[i] = arr[i - 1] * 2;
            if (i % 2 == 0) arr[i] += 1;
            else arr[i] -= 1;
            arr[i] = arr[i] %= 10007;
        }
        Write(arr[n]);

    }
}
